    exports.getLeafOfPath = (obj, path, create_object) ->
  parts = if Array.isArray path then path[..] else path.split '.'
  last = parts.pop()
  if parts.length > 0
    if create_object isnt false
      for part in parts
        obj = obj[part] ||= {}
    else
      for part in parts
        obj = obj[part]
        return [] if not obj
  [obj, last]
