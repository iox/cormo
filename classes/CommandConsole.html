class CommandConsole
  constructor: (argv) ->
    loads = []
    @program = program = new commander.Command('cormo')
    program.usage('console [options]')
      .option('-l, --load ', 'load specified module')
      .option('-d, --inspect-depth ', 'specify depth for util.inspect')
      .on('load', (path) -> loads.push path)
      .on('inspect-depth', (depth) => @inspect_depth = depth)
    program.help() if argv.indexOf('--help') >= 0 || argv.indexOf('-h') >= 0
    program.parse(argv)

    cwd = process.cwd()
    for load in loads
      console.log "Loading module '#{load}'..."
      try
        require load
      catch e
        console.log e.toString() if e.code isnt 'MODULE_NOT_FOUND'
        try
          require path.resolve cwd, load
        catch e
          console.log e.toString() if e.code isnt 'MODULE_NOT_FOUND'

  run: ->
    @runCoffee()

  runCoffee: ->
    cormo_console.startCoffee inspect_depth: @inspect_depth
    .on 'exit', ->
      process.exit 0
