class ModelQuery
  @_createQueryAndRun: (criteria, data, callback) ->
    query = new Query @
    query[criteria] data
    if typeof callback is 'function'
      query.exec callback
    query

  @_createOptionalQueryAndRun: (criteria, data, callback) ->
    if typeof data is 'function'
      @_createQueryAndRun criteria, null, data
    else
      @_createQueryAndRun criteria, data, callback
