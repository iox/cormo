          exports.pluralize = (str) ->
  if _uncountable_words.indexOf(str.toLowerCase()) is -1
    for rule in _plural_rules
      if not (rule[0] instanceof RegExp)
        rule[0] = new RegExp rule[0], 'i'
      if rule[0].test str
        return str.replace rule[0], rule[1]
  return str
